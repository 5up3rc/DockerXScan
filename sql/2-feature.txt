-- Table: public.feature

-- DROP TABLE public.feature;

CREATE TABLE public.feature
(
    id serial,
    namespace_id integer NOT NULL,
    name character varying(128) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT feature_pkey PRIMARY KEY (id),
    CONSTRAINT feature_namespace_id_name_key UNIQUE (namespace_id, name),
    CONSTRAINT feature_namespace_id_fkey FOREIGN KEY (namespace_id)
        REFERENCES public.namespace (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE public.feature
    OWNER to postgres;
